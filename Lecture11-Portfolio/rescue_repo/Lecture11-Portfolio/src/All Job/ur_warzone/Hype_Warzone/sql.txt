CREATE TABLE `xns_event` (
    `identifier` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8mb3_general_ci',
    `coords` LONGTEXT NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
    `status` LONGTEXT NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
    UNIQUE INDEX `identifier` (`identifier`) USING BTREE
)
COLLATE='utf8mb4_general_ci'
ENGINE=InnoDB
;
